         function groupby(d) {
             store = Array.from(d)
             alert("done")
             byid = d3.nest()
                 .key(function (d) { return d.AirlineID; })
                 .key(function (d) { return d.AirlineName; })
                 .rollup(function (v) {
                     return {
                         count: v.length,
                         country:
                             v.AirlineCountry

                     };
                 })
                 .entries(store);
             byid.forEach(function (d) {
                 d.AirlineID = d.key;
                 d.Airline = d.values[0];
                 d.AirlineName = d.Airline.key
                 d.count = d.Airline.value.count
             });
